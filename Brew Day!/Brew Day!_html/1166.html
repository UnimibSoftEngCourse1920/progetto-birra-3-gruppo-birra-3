161     	public Recipe featureWSIBT() {
162     		
163     		try {
164     			//inserisco tutte le ricette in un arraylist
165     			List<Recipe> recipes = extractRecipe();
166     			//creo hashmap per inserire le ricette possibili sulla base degli ingredienti
167     			Map<Integer, Double> recipeMax = new HashMap<>();
168     			//scorro l'arraylist e per ogni ricetta faccio il controllo sugli ingredienti
169     			for (int i = 0; i < recipes.size(); i++) {
170     				Recipe r = recipes.get(i);
171     				Map<String, Double> missingIngredients = r.computeMissingIngredients();
172     				double totIngredients = 0.0;
173     				//se sono presenti tutti gli ingredienti in storage calcolo gli ingredienti
174     				if(missingIngredients.isEmpty()) {
175     					Map<String, Double> ingredients = r.getIngredients();
176     					for(Entry<String, Double> j : ingredients.entrySet()) {
177     						totIngredients = totIngredients + j.getValue();
178     					}
179     					recipeMax.put(r.getId(), totIngredients);
180     				}			
181     			}
182     			
183     			//se esiste una ricetta plausibile
184     			if(!recipeMax.isEmpty()) {
185     				double max = 0.0;
186     				int id = -1;
187     				//cerco la ricetta che massimizza gli ingredienti utilizzati
188     				for(Entry<Integer, Double> r : recipeMax.entrySet()) {
189     					if(r.getValue() > max) {
190     						max = r.getValue();
191     					    id = r.getKey();
192     					}
193     				}
194     				//determino la ricetta da restituire
195     				for (int i = 0; i < recipes.size(); i++) {
196     					Recipe r = recipes.get(i);
197     					if(r.getId() == id) {
198     						return r;
199     					}
200     				}
201     			}
202     			throw new RecipeNotFoundException("Recipe WSIBT not found");
203     		}catch(RecipeNotFoundException e) {
204     			System.out.println(e.getMessage());
205     		}
206     		return null;
207     	}

209     	public void updateAllRecipesOnNewEquipment(double multiplier) {
210     		List<Recipe> recipes = extractRecipe();
211     		List<Recipe> newRecipes = new ArrayList<>();
212     		for(int i = 0; i<recipes.size(); i++) {
213     			Recipe r = recipes.get(i);
214     			Map<String, Double> ingredientsR = r.getIngredients();
215     			Map<String, Double> newIngredientsR = new HashMap<String, Double>();
216     			for (Entry<String,Double> i1 : ingredientsR.entrySet()) {
217     				double updateValue = Math.round(i1.getValue()*multiplier*100.0)/100.0;
218     				newIngredientsR.put(i1.getKey(), updateValue);
