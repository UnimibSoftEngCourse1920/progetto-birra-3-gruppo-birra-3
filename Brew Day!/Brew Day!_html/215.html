138     	public void actionPerformed(ActionEvent e) {
139     		switch(e.getActionCommand()) {
140     		case "Back":
141     			setVisible(false);
142     			new BrewWindow().setVisible(true);
143     			dispose();
144     			break;
145     		case "Add":
146     			setVisible(false);
147     			new AddOrModifyNoteWindow(brewId,0).setVisible(true);
148     			dispose();
149     			break;
150     		default:
151     			String[] tokens1 = e.getActionCommand().split("/");
152     			int noteId1 = Integer.parseInt(tokens1[0]);
153     			String command = tokens1[1];
154     			int row = Integer.parseInt(tokens1[2]);
155     
156     			switch(command) {
157     			case "Modify":
158     				setVisible(false);
159     				new AddOrModifyNoteWindow(brewId,noteId1).setVisible(true);
160     				dispose();
161     				break;
162     			case "Delete":
163     				brewController.deleteNote(brewId, noteId1);
164     				JTable table1 = (JTable)e.getSource();
165     				((DefaultTableModel)table1.getModel()).removeRow(row);
166     				break;
167     			default:
168     			}
169     		}
