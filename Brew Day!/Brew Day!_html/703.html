125     	@Override
126     	public void actionPerformed(ActionEvent e) {
127     		File f;
128     	
129     		switch(e.getActionCommand()) {
130     			case "Show Equipment":
131     				f = new File(filepath);
132     				if(!f.exists()) {
133     					JOptionPane.showMessageDialog(this, "You have no equipment to show");
134     				}
135     				else {
136     					EquipmentController equipmentController = EquipmentController.getInstance();
137     					int size = equipmentController.extractEquipment().getInstruments().size();
138     					if(size == 0) {
139     						JOptionPane.showMessageDialog(this, "You have no instruments in your equipment");
140     					}
141     					else {
142     						ShowEquipmentWindow showEqWin = new ShowEquipmentWindow();
143     						showEqWin.setVisible(true);
144     						dispose();
145     					}
146     				}
147     				break;
148     			case "Create Equipment":
149     				f = new File(filepath);
150     				if(!f.exists()) {
151     					CreateEquipmentWindow createEqWin = new CreateEquipmentWindow();
152     					createEqWin.setVisible(true);
153     					dispose();
154     				}
155     				else {
156     					EquipmentController equipmentController = EquipmentController.getInstance();
157     					int size = equipmentController.extractEquipment().getInstruments().size();
158     					if(size == 0) {
159     						CreateEquipmentWindow createEqWin = new CreateEquipmentWindow();
160     						createEqWin.setVisible(true);
161     						dispose();
162     					}
163     					else {
164     						JOptionPane.showMessageDialog(this, "You already have an equipment");
165     					}
166     				}
167     				break;
168     			case "Modify Equipment":
169     				f = new File(filepath);
170     				if(!f.exists()) {
171     					JOptionPane.showMessageDialog(this, "You have no equipment");
172     				}
173     				else {
174     					EquipmentController equipmentController = EquipmentController.getInstance();
175     					int size = equipmentController.extractEquipment().getInstruments().size();
176     					if(size == 0) {
177     						JOptionPane.showMessageDialog(this, "You have no instruments in your equipment to modify");
178     					}
179     					else {
180     						ModifyEquipmentWindow modifyEqWin = new ModifyEquipmentWindow();
181     						modifyEqWin.setVisible(true);
182     						dispose();
183     					}
184     				}
185     				break;
186     			case "Insert New Instrument":
187     				f = new File(filepath);
188     				if(!f.exists()) {
189     					JOptionPane.showMessageDialog(this, "You should create a new equipment first");
190     				}
191     				else {
192     					InsertNewInstrumentWindow insertNewInstrumentWin = new InsertNewInstrumentWindow();
193     					insertNewInstrumentWin.setVisible(true);
194     					dispose();
195     				}
196     				break;
197     			case "Delete Instrument":
198     				EquipmentController equipmentController = EquipmentController.getInstance();
199     				f = new File(filepath);
200     				if(!f.exists()) {
201     					JOptionPane.showMessageDialog(this, "You have no equipment to delete");
202     				}else {
203     						int size = equipmentController.extractEquipment().getInstruments().size();
204     						if(size != 0) {
205     							DeleteInstrumentWindow deleteInstrumentWin = new DeleteInstrumentWindow();
206     							deleteInstrumentWin.setVisible(true);
207     							dispose();
208     						}else {
209     						JOptionPane.showMessageDialog(this, "You have no equipment to delete");
210     					}
211     				}
212     				break;
213     			case "Back":
214     				MainWindow mainWin = new MainWindow();
215     				mainWin.setVisible(true);
216     				dispose();
217     				break;
218     			default:
219     		}
