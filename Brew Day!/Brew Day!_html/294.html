97     		if(missingIngredients.isEmpty()) {
98     			Date currentDate = new Date(System.currentTimeMillis());

101     			BrewController brewController = BrewController.getInstance();

103     			//subtract ingredients from storage
104     			Storage storage = Storage.getInstance();
105     			Map<String,Double> storageIngredients = storage.getIngredients();
106     			for (Entry<String,Double> i : this.ingredients.entrySet()) {
107     				double sIngredientValue = storageIngredients.get(i.getKey()).doubleValue();
108     				storageIngredients.replace(i.getKey(),sIngredientValue-i.getValue());
109     			}
110     			
111     			StorageController storageController = StorageController.getInstance();
112     			storageController.update(storageIngredients);
113     			return b;
114     		}
115     		else {
116     			StringBuilder missingAlert = new StringBuilder();
117     			missingAlert.append("Some ingredients are missing from your storage! \nYou should buy:");
118     			for(Entry<String, Double> i : missingIngredients.entrySet()) {
119     				missingAlert.append("\n" + i.getKey() + " : " + missingIngredients.get(i.getKey()));

154     	public Map<String, Double> computePercentage(Map<String, Double> ingredients){
155     		if(ingredients != null) {
156     			Map<String, Double> percentageIngredients = new HashMap<>();
157     			Double totalGrams = null;
158     			for (Entry<String,Double> i : this.ingredients.entrySet()) {
159     				totalGrams = i.getValue();
160     			}
161     			for (Entry<String,Double> i : this.ingredients.entrySet()) {
162     				if(totalGrams != null && totalGrams.compareTo(0.0) != 0)
163     					percentageIngredients.put(i.getKey(), i.getValue()/totalGrams);
164     			}
165     			return percentageIngredients;
166     		}
167     		return null;
168     	}
