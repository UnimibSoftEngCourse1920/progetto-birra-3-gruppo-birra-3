95     	private Map<String, Double> updateInstruments(){
96     		try {
97     			Map<String, Double> instruments = new HashMap<>();
98     			for (int i = 0; i < table.getRowCount(); i++) {
99     				String instrumentCapacity = table.getValueAt(i, 1).toString();
100     				if (!instrumentCapacity.matches("[0-9]+.{0,1}[0-9]*")) {
101     					throw new IllegalArgumentException();
102     				}
103     				instruments.put(table.getValueAt(i, 0).toString(), Double.parseDouble(instrumentCapacity));
104     			}
105     			return instruments;
106     		} catch (IllegalArgumentException e) {
107     			JOptionPane.showMessageDialog(this,"Insert only positive numbers, separated by dot");
108     			return null;
109     		}
110     	}

118     		else {
119     			if (table.isEditing())
120     			    table.getCellEditor().stopCellEditing();
121     			
122     			if(updateInstruments() != null) {
123     				EquipmentController equipmentController = EquipmentController.getInstance();
124     				equipmentController.update(updateInstruments());
125     				
126     				new ShowEquipmentWindow().setVisible(true);
