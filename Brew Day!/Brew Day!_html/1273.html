230     	public void actionPerformed(ActionEvent e) {
231     		switch(e.getActionCommand()) {
232     		case "View recipes":
233     			setVisible(false);
234     			new RecipeWindow().setVisible(true);
235     			break;
236     		case "View brews":
237     			File f = new File(System.getProperty("user.dir") + "\\src\\Files\\Brew.txt");
238     
239     			if (!f.exists() || BrewController.getInstance().extractBrew().isEmpty()) {
240     				JOptionPane.showMessageDialog(this, "No brews to show");
241     			} else {
242     				setVisible(false);
243     				new BrewWindow().setVisible(true);
244     				dispose();
245     			}
246     			break;
247     		case "View equipment":
248     			setVisible(false);
249     			new EquipmentWindow().setVisible(true);
250     			break;
251     		case "View storage":
252     			setVisible(false);
253     			new StorageWindow().setVisible(true);
254     			break;
255     		case "Brew it!":
256     			if(brewIt) {
257     				RecipeController recipeController = RecipeController.getInstance();
258     				Recipe wsibtRecipe = getWSIBT();
259     				recipeController.createBrew(wsibtRecipe.getId());
260     
261     				BrewWindow brewWin = new BrewWindow();
262     				brewWin.setVisible(true);
263     				dispose();
264     			}
265     			else
266     				JOptionPane.showMessageDialog(this, "You can't Brew it!");
267     			break;
268     		case "Reset":
269     			recipeController.deleteFile();
270     			BrewController.getInstance().deleteFile();
271     			StorageController.getInstance().deleteFile();
272     			EquipmentController.getInstance().deleteFile();
273     			try (FileOutputStream fos = new FileOutputStream(System.getProperty("user.dir") + "\\src\\Files\\CounterId.txt");
274     				DataOutputStream dos = new DataOutputStream(fos)) {
275     				dos.writeInt(0); 
276     			} catch (IOException ioe) {
277     				System.out.println("IOException : " + ioe);
278     			}
279     		default:
280     		}
281     	}
