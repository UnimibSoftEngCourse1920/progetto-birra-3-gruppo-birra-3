112     	public Brew createBrew(){
113     		missingIngredients = this.computeMissingIngredients();
114     		
115     		//if no ingredients are missing from the storage the brew is created and stored
116     		if(missingIngredients.isEmpty()) {
117     			Date currentDate = new Date(System.currentTimeMillis());
118     			Brew b = new Brew(this, currentDate, Equipment.getInstance().getCapacity());
119     			countBrew++;
120     			BrewController brewController = BrewController.getInstance();
121     			brewController.store(b);
122     			
123     			//subtract used ingredients from the storage
124     			Storage storage = Storage.getInstance();
125     			Map<String,Double> storageIngredients = storage.getIngredients();
126     			for (Entry<String,Double> i : this.ingredients.entrySet()) {
127     				double sIngredientValue = storageIngredients.get(i.getKey()).doubleValue();
128     				storageIngredients.replace(i.getKey(), Math.round((sIngredientValue-i.getValue())*100)/100.0);
129     			}
130     			
131     			//updates the storage instance and file
132     			StorageController storageController = StorageController.getInstance();
133     			storageController.update(storageIngredients);
134     			return b;
135     		}
136     		//if some ingredients are missing the an alert is shown
137     		else {
138     			StringBuilder missingAlert = new StringBuilder();
139     			missingAlert.append("Some ingredients are missing from your storage! \nYou should buy:");
140     			for(Entry<String, Double> i : missingIngredients.entrySet()) {
141     				missingAlert.append("\n" + i.getKey() + " : " +  Math.round(missingIngredients.get(i.getKey())*100)/100);
142     			}
143     			return null;
144     		}
145     	}

182     	 * Returns the Map of the ingredients quantities expressed as
183     	 * a percentage of the total sum of their quantities
184     	 */
185     	public Map<String, Double> computePercentage(Map<String, Double> ingredients){
186     		if(ingredients != null) {
187     			Map<String, Double> percentageIng = new HashMap<>();
188     			Double totalGrams = 0.0;
189     			for (Entry<String,Double> i : this.ingredients.entrySet()) {
190     				totalGrams += i.getValue();
191     			}
192     			for (Entry<String,Double> i : this.ingredients.entrySet()) {
193     				if(totalGrams.compareTo(0.0) != 0)
194     					percentageIng.put(i.getKey(), i.getValue()/totalGrams);
195     			}
196     			return percentageIng;
197     		}
198     		return null;
199     	}
