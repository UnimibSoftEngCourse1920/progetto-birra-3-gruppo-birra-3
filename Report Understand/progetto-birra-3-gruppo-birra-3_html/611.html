195     	 * Returns the recipe that maximize the usage of the available ingredients, 
196     	 * if there aren't recipes that can be brewed or there aren't recipes at all
197     	 * an exception is thrown
198     	 */
199     	public Recipe featureWSIBT() {
200     		try {
201     			//puts the recipes in a list
202     			List<Recipe> recipes = extractRecipe();
203     			//creates a map to contain the recipes for which there aren't missing ingredients
204     			Map<Integer, Double> recipeMax = new HashMap<>();
205     			//browses the list and computes the missing ingredients for each recipe
206     			for (int i = 0; i < recipes.size(); i++) {
207     				Recipe r = recipes.get(i);
208     				Map<String, Double> missingIngredients = r.computeMissingIngredients();
209     				double totIngredients = 0.0;
210     				//if there aren't missing ingredients the total sum of the ingredients is saved in the map along with the recipe
211     				if(missingIngredients.isEmpty()) {
212     					Map<String, Double> ingredients = r.getIngredients();
213     					for(Entry<String, Double> j : ingredients.entrySet()) {
214     						totIngredients = totIngredients + j.getValue();
215     					}
216     					recipeMax.put(r.getId(), totIngredients);
217     				}			
218     			}
219     			
220     			//if there is a recipe for which there aren't missing ingredients
221     			if(!recipeMax.isEmpty()) {
222     				double max = 0.0;
223     				int id = -1;
224     				//looks for the recipe that maximize the ingredients usage
225     				for(Entry<Integer, Double> r : recipeMax.entrySet()) {
226     					if(r.getValue() > max) {
227     						max = r.getValue();
228     					    id = r.getKey();
229     					}
230     				}
231     				//returns the right recipe
232     				for (int i = 0; i < recipes.size(); i++) {
233     					Recipe r = recipes.get(i);
234     					if(r.getId() == id) {
235     						return r;
236     					}
237     				}
238     			}
239     			throw new RecipeNotFoundException("Recipe WSIBT not found");
240     		}catch(RecipeNotFoundException e) {
241     			System.out.println(e.getMessage());
242     		}
243     		return null;
244     	}

256     			for (Entry<String,Double> i1 : ingredientsR.entrySet()) {
257     				
258     				//computes the new ingredient quantity using the multiplier
