66     		try {

76     			//updates the recipes

79     			RecipeController recipeController = RecipeController.getInstance();
80     			recipeController.updateAllRecipesOnNewEquipment(ingredientsMultiplier);
81     			
82     		}catch(NullInstrumentsException e) {
83     			System.out.println(e.getMessage());
