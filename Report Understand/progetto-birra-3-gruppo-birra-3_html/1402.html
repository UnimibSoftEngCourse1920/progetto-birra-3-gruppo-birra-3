222     	/*
223     	 * Returns the ingredients map using the data inserted in the given table 
224     	 * located in the given window. Shows an alert message and returns null if the input format
225     	 * is not correct or if nothing has been inserted
226     	 */
227     	public static Map<String, Double> createIngredients(JFrame window,JTable table){
228     		try {
229     			Map<String, Double> ingredients = new HashMap<>();
230     			
231     			for (int i = 0; i < table.getRowCount(); i++) {
232     				String ingredientName = table.getValueAt(i, 0).toString();
233     				String ingredientQuantity = table.getValueAt(i, 1).toString();
234     				if (!ingredientName.matches("[a-zA-Z_]+")) {
235     					throw new IllegalArgumentException();
236     				}
237     				else if(!(ingredientQuantity.equals("0.0") || ingredientQuantity.equals("0") || ingredientQuantity.equals(""))) {
238     					ingredients.put(table.getValueAt(i, 0).toString(), fromStringToDouble(ingredientQuantity));
239     				}
240     			}
241     			if(ingredients.isEmpty()) {
242     				throw new NullInputException();
243     			}
244     			return ingredients;
245     		} catch (NumberFormatException e) {
246     			JOptionPane.showMessageDialog(window,"Insert only positive numbers in quantity field, separated by dot (e.g. Sugar 10.50)");
247     			return null;
248     		} catch (IllegalArgumentException e) {
249     			JOptionPane.showMessageDialog(window,"Insert only string in ingredient name field");
250     			return null;
251     		} catch (NullInputException e) {
252     			JOptionPane.showMessageDialog(window,"Insert at least an ingredient");
253     			return null;
254     		}
255     	}

257     	/*
258     	 * Parses the given string str into a Double and returns it.
259     	 * Throws a NumberFormatException if the number is negative
260     	 */
261     	public static double fromStringToDouble(String str) {
262     		if (str.contains("-")) {
263     			throw new NumberFormatException();
264     		} 
265     		return Double.parseDouble(str);
266     	}
